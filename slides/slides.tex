int position = hash(keyaddr, keylen) & (HASHSIZE-1);
struct hashnode *l; l = ht[position];
while (l != NULL) {
	if (keylen == l->keylen &&
		memcmp(keyaddr, l->keyaddr, keylen) == 0)
		return l->value;
	if (position <= HASHSIZE)
		l = ht[++position];
	else
		break;
}	
\end{lstlisting}}
